def get_counts(threshold):
    cbad = temp_readings[(temp_readings < mean - threshold * std_dev) | (temp_readings > mean + threshold * std_dev)]
    cgood = temp_readings[(temp_readings > mean - threshold * std_dev) & (temp_readings < mean + threshold * std_dev)]
    tneg = len(cbad[np.isin(cbad, btempreadings)])
    tpos = len(cgood[np.isin(cgood, gtempreadings)])
    fneg = len(cbad[np.isin(cbad, gtempreadings)])
    fpos = len(cgood[np.isin(cgood, btempreadings)])
    return np.array([tneg, tpos, fneg, fpos])
thresholds = np.linspace(0, 10, 200)
with Pool() as p:
    results = np.transpose(np.array(p.map(get_counts, thresholds)))

def plot_counts(results, log_scale=False, draw_result_line=False):
    plt.title('good and bad counts vs threshold')
    plt.xlabel('sigma threshold')
    plt.ylabel('count')
    plt.plot(thresholds, results[0])
    plt.plot(thresholds, results[1])
    plt.plot(thresholds, results[2])
    plt.plot(thresholds, results[3])
    if draw_result_line:
        max_tthr = thresholds[np.argmax(results[0] + results[1])]
        min_fthr = thresholds[np.argmin(results[2] + results[3])]
        print(f"Max true: {max_tthr} sigma, Min false: {min_fthr} sigma")
        plt.axvline(max_tthr, 0, 1, color='k', linestyle='--')
        plt.axvline(min_fthr, 0, 1, color='k', linestyle='--')
    if log_scale:
        plt.yscale('log')
    plt.xlim(0, 10)
    plt.xticks(range(0, 11))
    plt.legend(['True negatives', 'True positives', 'False negatives', 'False positives'])    
    if draw_result_line:
        return max_tthr, min_fthr

plot_counts(results, log_scale=False, draw_result_line=False)
